157static void pub_timer_cb(void *arg)
158{
159 if (!s_vnd_model || !s_vnd_model->pub) return;
160
161 esp_err_t err;
162 int16_t temp_x100 = 0;
163 uint16_t rh_x100 = 0;
164 uint32_t raw_red = 0;
165 uint32_t raw_ir = 0;
166 uint8_t heart_rate = 0;
167 uint8_t spo2 = 0;
168 uint16_t eco2 = 0;
169 uint16_t tvoc = 0;
170
171#if ENABLE_AHT20
172 // Read temperature and humidity from AHT20
173 err = aht20_read_x100(&temp_x100, &rh_x100);
174 if (err != ESP_OK) {
175 ESP_LOGW(TAG, "AHT20 read failed: %d", err);
176 temp_x100 = 0;
177 rh_x100 = 0;
178 }
179#endif
180
181#if ENABLE_MAX30101
182 // Get latest raw values from MAX30101
183 max30101_get_raw_values(&raw_red, &raw_ir);
184
185 // Get computed HR and SpO2 from MAX30101
186 heart_rate = max30101_get_heart_rate();
187 spo2 = max30101_get_spo2();
188#endif
189
190#if ENABLE_SGP30
191 // Use cached values from 1Hz timer (SGP30 requires continuous 1Hz measurements)
192 eco2 = cached_eco2;
193 tvoc = cached_tvoc;
194 ESP_LOGI(TAG, "SGP30: eCO2=%d ppm, TVOC=%d ppb", eco2, tvoc);
195#endif
196
197 // Pack payload: temp(2) + humidity(2) + red(4) + ir(4) + HR(1) + SpO2(1) + eco2(2) + tvoc(2) = 18 bytes
198 uint8_t payload[18] = {0};
199 payload[0] = (uint8_t)(temp_x100 & 0xFF);
200 payload[1] = (uint8_t)((temp_x100 >> 8) & 0xFF);
201 payload[2] = (uint8_t)(rh_x100 & 0xFF);
202 payload[3] = (uint8_t)((rh_x100 >> 8) & 0xFF);
203 payload[4] = (uint8_t)(raw_red & 0xFF);
204 payload[5] = (uint8_t)((raw_red >> 8) & 0xFF);
205 payload[6] = (uint8_t)((raw_red >> 16) & 0xFF);
206 payload[7] = (uint8_t)((raw_red >> 24) & 0xFF);
207 payload[8] = (uint8_t)(raw_ir & 0xFF);
208 payload[9] = (uint8_t)((raw_ir >> 8) & 0xFF);
209 payload[10] = (uint8_t)((raw_ir >> 16) & 0xFF);
210 payload[11] = (uint8_t)((raw_ir >> 24) & 0xFF);
211 payload[12] = heart_rate;
212 payload[13] = spo2;
213 payload[14] = (uint8_t)(eco2 & 0xFF);
214 payload[15] = (uint8_t)((eco2 >> 8) & 0xFF);
215 payload[16] = (uint8_t)(tvoc & 0xFF);
216 payload[17] = (uint8_t)((tvoc >> 8) & 0xFF);
217
218 err = esp_ble_mesh_model_publish(
219 s_vnd_model,
220 OP_SENSOR_STATUS,
221 sizeof(payload),
222 payload,
223 ROLE_NODE
224 );
225
226 if (err != ESP_OK) {
227 ESP_LOGW(TAG, "Publish failed: %d", err);
228 } else {
229 ESP_LOGI(TAG, "Published: Temp=%d.%02d°C RH=%d.%02d%% RED=%lu IR=%lu HR=%ubpm SpO2=%d%% eCO2=%d ppm TVOC=%d ppb",
230 temp_x100/100, abs(temp_x100%100), rh_x100/100, rh_x100%100,
231 (unsigned long)raw_red, (unsigned long)raw_ir, heart_rate, spo2, eco2, tvoc);
232 }
233}

248static void example_ble_mesh_provisioning_cb(esp_ble_mesh_prov_cb_event_t event,
249 esp_ble_mesh_prov_cb_param_t *param)
250{
251 switch (event) {
252 case ESP_BLE_MESH_PROV_REGISTER_COMP_EVT:
253 ESP_LOGI(TAG, "ESP_BLE_MESH_PROV_REGISTER_COMP_EVT, err_code %d", param->prov_register_comp.err_code);
254 break;
255 case ESP_BLE_MESH_NODE_PROV_ENABLE_COMP_EVT:
256 ESP_LOGI(TAG, "ESP_BLE_MESH_NODE_PROV_ENABLE_COMP_EVT, err_code %d", param->node_prov_enable_comp.err_code);
257 break;
258 case ESP_BLE_MESH_NODE_PROV_LINK_OPEN_EVT:
259 ESP_LOGI(TAG, "ESP_BLE_MESH_NODE_PROV_LINK_OPEN_EVT, bearer %s",
260 param->node_prov_link_open.bearer == ESP_BLE_MESH_PROV_ADV ? "PB-ADV" : "PB-GATT");
261 break;
262 case ESP_BLE_MESH_NODE_PROV_LINK_CLOSE_EVT:
263 ESP_LOGI(TAG, "ESP_BLE_MESH_NODE_PROV_LINK_CLOSE_EVT, bearer %s",
264 param->node_prov_link_close.bearer == ESP_BLE_MESH_PROV_ADV ? "PB-ADV" : "PB-GATT");
265
266 // After provisioning, device should now be advertising as GATT proxy
267 // If using PB-ADV, you need a GATT proxy relay to reach this device
268 // If using PB-GATT, the connection should persist for configuration
269 if (param->node_prov_link_close.bearer == ESP_BLE_MESH_PROV_ADV) {
270 ESP_LOGI(TAG, "Provisioned via PB-ADV - device will advertise as GATT proxy");
271 } else {
272 ESP_LOGI(TAG, "Provisioned via PB-GATT - persistent GATT connection may be available");
273 }
274 break;
275 case ESP_BLE_MESH_NODE_PROV_COMPLETE_EVT:
276 ESP_LOGI(TAG, "ESP_BLE_MESH_NODE_PROV_COMPLETE_EVT");
277 ESP_LOGI(TAG, "Device provisioned: net_idx=0x%04x, addr=0x%04x",
278 param->node_prov_complete.net_idx, param->node_prov_complete.addr);
279
280 prov_complete(param->node_prov_complete.net_idx, param->node_prov_complete.addr,
281 param->node_prov_complete.flags, param->node_prov_complete.iv_index);
282
283 // TODO: set this to your actual vendor model address:
284 s_vnd_model = &vnd_models[0]; // <- change if your model array name differs

298 break;

300 break;
301 }

338static void example_ble_mesh_custom_model_cb(esp_ble_mesh_model_cb_event_t event,
339 esp_ble_mesh_model_cb_param_t *param)
340{
341 switch (event) {
342 case ESP_BLE_MESH_MODEL_OPERATION_EVT:
343 if (param->model_operation.opcode == ESP_BLE_MESH_VND_MODEL_OP_SEND) {
344 uint16_t tid = *(uint16_t *)param->model_operation.msg;
345 ESP_LOGI(TAG, "Recv 0x%06" PRIx32 ", tid 0x%04x", param->model_operation.opcode, tid);
346 esp_err_t err = esp_ble_mesh_server_model_send_msg(&vnd_models[0],

348 sizeof(tid), (uint8_t *)&tid);
349 if (err) {
350 ESP_LOGE(TAG, "Failed to send message 0x%06x", ESP_BLE_MESH_VND_MODEL_OP_STATUS);
351 }
352 }
353 break;
354 case ESP_BLE_MESH_MODEL_SEND_COMP_EVT:
355 if (param->model_send_comp.err_code) {
356 ESP_LOGE(TAG, "Failed to send message 0x%06" PRIx32, param->model_send_comp.opcode);
357 break;
358 }
359 ESP_LOGI(TAG, "Send 0x%06" PRIx32, param->model_send_comp.opcode);
360 break;
361 default:
362 break;
363 }

369static esp_err_t ble_mesh_init(void)
370{
371 esp_err_t err;
372
373 esp_ble_mesh_register_prov_callback(example_ble_mesh_provisioning_cb);
374 esp_ble_mesh_register_config_server_callback(example_ble_mesh_config_server_cb);
375 esp_ble_mesh_register_custom_model_callback(example_ble_mesh_custom_model_cb);
376
377 err = esp_ble_mesh_init(&provision, &composition);
378 if (err != ESP_OK) {
379 ESP_LOGE(TAG, "Failed to initialize mesh stack");
380 return err;
381 }
382
383 err = esp_ble_mesh_set_unprovisioned_device_name(BLE_MESH_DEVICE_NAME);
384 if (err != ESP_OK) {
385 ESP_LOGW(TAG, "Failed to set unprovisioned name: %d", err);
386 } else {
387 ESP_LOGI(TAG, "Unprovisioned name set to: %s", BLE_MESH_DEVICE_NAME);
388 }
389
390 err = esp_ble_mesh_node_prov_enable((esp_ble_mesh_prov_bearer_t)(ESP_BLE_MESH_PROV_ADV | ESP_BLE_MESH_PROV_GATT));
391 if (err != ESP_OK) {
392 ESP_LOGE(TAG, "Failed to enable mesh node");
393 return err;
394 }
395
396 board_led_operation(LED_G, LED_ON);
397
398 ESP_LOGI(TAG, "BLE Mesh Node initialized");
399
400 return ESP_OK;
401}

406void app_main(void)
407{
408 esp_err_t err;
409
410 ESP_LOGI(TAG, "=== BLE Mesh Sensor Server Starting ===");
411
412 // Initialize NVS
413 err = nvs_flash_init();
414 if (err == ESP_ERR_NVS_NO_FREE_PAGES) {
415 ESP_ERROR_CHECK(nvs_flash_erase());
416 err = nvs_flash_init();
417 }
418 ESP_ERROR_CHECK(err);
419
420 // Initialize board peripherals
421 board_init();
422
423 // Initialize sensors
424 ESP_LOGI(TAG, "Initializing sensors...");
425#if ENABLE_AHT20
426 ESP_LOGI(TAG, "Initializing AHT20 (temp/humidity)");
427 ESP_ERROR_CHECK(aht20_init());
428#endif
429
430#if ENABLE_MAX30101
431 ESP_LOGI(TAG, "Initializing MAX30101 (heart rate/SpO2)");
432 ESP_ERROR_CHECK(max30101_init());
433 ESP_ERROR_CHECK(max30101_start_sampling());
434#endif
435
436#if ENABLE_SGP30
437 ESP_LOGI(TAG, "Initializing SGP30 (air quality)");
438 ESP_ERROR_CHECK(sgp30_init());
439
440 // Start 1Hz timer immediately - SGP30 requires continuous 1Hz measurements
441 const esp_timer_create_args_t sgp30_targs = {
442 .callback = &sgp30_timer_cb,
443 .name = "sgp30_1hz"
444 };
445 ESP_ERROR_CHECK(esp_timer_create(&sgp30_targs, &sgp30_timer));
446 ESP_ERROR_CHECK(esp_timer_start_periodic(sgp30_timer, 1 * 1000 * 1000)); // 1s
447 ESP_LOGI(TAG, "SGP30 1Hz baseline timer started");
448#endif
449
450 // Initialize Bluetooth
451 err = bluetooth_init();
452 if (err) {
453 ESP_LOGE(TAG, "Bluetooth init failed (err %d)", err);
454 return;
455 }
456
457 // Set the Bluetooth device name (for GAP)
458 err = esp_ble_gap_set_device_name(BLE_MESH_DEVICE_NAME);
459 if (err != ESP_OK) {
460 ESP_LOGW(TAG, "Failed to set BLE GAP device name: %d", err);
461 }
462
463 // Get device UUID
464 ble_mesh_get_dev_uuid(dev_uuid);
465
466 // Initialize BLE Mesh subsystem
467 err = ble_mesh_init();
468 if (err) {
469 ESP_LOGE(TAG, "BLE Mesh init failed (err %d)", err);
470 }
471
472 ESP_LOGI(TAG, "=== Initialization Complete ===");
473}
