93static esp_err_t max30101_read_sample(uint32_t *red, uint32_t *ir)
94{
95 uint8_t d[6];
96 esp_err_t err = max30101_read(REG_FIFO_DATA, d, sizeof(d));
97 if (err) return err;
98
99 // Parse 18-bit values (3 bytes each for RED and IR)
100 uint32_t raw_red = ((uint32_t)d[0] << 16) | ((uint32_t)d[1] << 8) | d[2];
101 uint32_t raw_ir = ((uint32_t)d[3] << 16) | ((uint32_t)d[4] << 8) | d[5];
102
103 raw_red &= 0x3FFFF; // 18-bit mask
104 raw_ir &= 0x3FFFF;
105
106 *red = raw_red;
107 *ir = raw_ir;
108 return ESP_OK;
109}

125// Heart Rate Calculation (Peak Detection Algorithm)
129static uint8_t calculate_heart_rate(void)
130{
131 if (s_ppg_buffer.count < SAMPLE_BUFFER_SIZE) {
132 return 0; // Not enough samples
133 }
134
135 // Find min/max in IR signal
136 uint32_t min_val = UINT32_MAX, max_val = 0;
137 for (int i = 0; i < SAMPLE_BUFFER_SIZE; i++) {
138 if (s_ppg_buffer.ir_samples[i] < min_val) min_val = s_ppg_buffer.ir_samples[i];
139 if (s_ppg_buffer.ir_samples[i] > max_val) max_val = s_ppg_buffer.ir_samples[i];
140 }
141
142 if (max_val <= min_val || max_val < 10000) {
143 return 0; // No valid signal
144 }
145
146 uint32_t threshold = min_val + (uint32_t)((max_val - min_val) * PEAK_THRESHOLD_RATIO);
147
148 // Peak detection
149 uint32_t last_peak = 0;
150 bool above_threshold = false;
151
152 for (int i = 1; i < SAMPLE_BUFFER_SIZE - 1; i++) {
153 uint32_t curr = s_ppg_buffer.ir_samples[i];
154 uint32_t prev = s_ppg_buffer.ir_samples[i - 1];
155 uint32_t next = s_ppg_buffer.ir_samples[i + 1];
156
157 if (curr > threshold && curr >= prev && curr >= next) {
158 if (!above_threshold && (last_peak == 0 || (i - last_peak) >= MIN_PEAK_DISTANCE)) {
159 if (last_peak > 0) {
160 uint32_t interval = i - last_peak;
161 s_ppg_buffer.peak_intervals[s_ppg_buffer.interval_index] = interval;
162 s_ppg_buffer.interval_index = (s_ppg_buffer.interval_index + 1) % 5;
163 if (s_ppg_buffer.interval_count < 5) s_ppg_buffer.interval_count++;
164 }
165 last_peak = i;
166 above_threshold = true;
167 }
168 } else if (curr < threshold) {
169 above_threshold = false;
170 }
171 }
172
173 // Calculate average interval
174 if (s_ppg_buffer.interval_count < 2) {
175 return 0;
176 }
177
178 uint32_t avg_interval = 0;
179 for (int i = 0; i < s_ppg_buffer.interval_count; i++) {
180 avg_interval += s_ppg_buffer.peak_intervals[i];
181 }
182 avg_interval /= s_ppg_buffer.interval_count;
183
184 if (avg_interval == 0) return 0;
185
186 // Convert to BPM (100 samples/sec)
187 uint32_t bpm = (100 * 60) / avg_interval;
188
189 // Sanity check (30-220 BPM)
190 if (bpm < 30 || bpm > 220) return 0;
191
192 return (uint8_t)bpm;
193}

195// SpO2 Calculation (AC/DC Ratio Method)
199static uint8_t calculate_spo2(void)
200{
201 if (s_ppg_buffer.count < SAMPLE_BUFFER_SIZE) {
202 return 0;
203 }
204
205 // Calculate AC and DC components for RED and IR
206 uint32_t red_min = UINT32_MAX, red_max = 0;
207 uint32_t ir_min = UINT32_MAX, ir_max = 0;
208 uint64_t red_sum = 0, ir_sum = 0;
209
210 for (int i = 0; i < SAMPLE_BUFFER_SIZE; i++) {
211 uint32_t red = s_ppg_buffer.red_samples[i];
212 uint32_t ir = s_ppg_buffer.ir_samples[i];
213
214 if (red < red_min) red_min = red;
215 if (red > red_max) red_max = red;
216 if (ir < ir_min) ir_min = ir;
217 if (ir > ir_max) ir_max = ir;
218
219 red_sum += red;
220 ir_sum += ir;
221 }
222
223 uint32_t red_dc = (uint32_t)(red_sum / SAMPLE_BUFFER_SIZE);
224 uint32_t ir_dc = (uint32_t)(ir_sum / SAMPLE_BUFFER_SIZE);
225 uint32_t red_ac = red_max - red_min;
226 uint32_t ir_ac = ir_max - ir_min;
227
228 // Check for valid signal
229 if (red_dc < 10000 || ir_dc < 10000 || red_ac == 0 || ir_ac == 0) {
230 return 0;
231 }
232
233 // Calculate R value: (AC_red/DC_red) / (AC_ir/DC_ir)
234 float r = ((float)red_ac / (float)red_dc) / ((float)ir_ac / (float)ir_dc);
235
236 // Empirical formula (adjust based on calibration)
237 // SpO2 = 110 - 25*R
238 float spo2 = 110.0f - 25.0f * r;
239
240 // Sanity check (70-100%)
241 if (spo2 < 70.0f || spo2 > 100.0f) {
242 return 0;
243 }
244
245 return (uint8_t)spo2;
246}

248// Background Sampling Task
252static void ppg_sampling_task(void *arg)
253{
254 ESP_LOGI(TAG, "PPG sampling task started");
255
256 while (1) {
257 uint32_t red = 0, ir = 0;
258 esp_err_t err = max30101_read_sample(&red, &ir);
259
260 if (err == ESP_OK && ir > 1000) { // Valid signal check
261 ppg_add_sample(red, ir);
262
263 // Store latest raw values
264 s_latest_raw_red = red;
265 s_latest_raw_ir = ir;
266
267 // Update HR and SpO2 every 10 samples (~100ms)
268 static uint8_t sample_count = 0;
269 if (++sample_count >= 10) {
270 sample_count = 0;
271 s_latest_hr = calculate_heart_rate();
272 s_latest_spo2 = calculate_spo2();
273
274 if (s_latest_hr > 0 || s_latest_spo2 > 0) {
275 ESP_LOGD(TAG, "HR=%u SpO2=%u%%", s_latest_hr, s_latest_spo2);
276 }
277 }
278 }
279
280 vTaskDelay(pdMS_TO_TICKS(10)); // 100Hz sampling
281 }
282}

288esp_err_t max30101_init(void)
289{
290 // Initialize I2C bus
291 i2c_master_bus_config_t bus_cfg = {
292 .clk_source = I2C_CLK_SRC_DEFAULT,
293 .i2c_port = MAX30101_I2C_PORT,
294 .scl_io_num = MAX30101_SCL_GPIO,
295 .sda_io_num = MAX30101_SDA_GPIO,
296 .glitch_ignore_cnt = 7,
297 .flags.enable_internal_pullup = true,
298 };
299
300 ESP_RETURN_ON_ERROR(i2c_new_master_bus(&bus_cfg, &s_max30101_bus), TAG, "I2C bus init failed");
301
302 // Add MAX30101 device
303 i2c_device_config_t dev_cfg = {
304 .dev_addr_length = I2C_ADDR_BIT_LEN_7,
305 .device_address = MAX30101_ADDR,
306 .scl_speed_hz = MAX30101_FREQ_HZ,
307 };
308 ESP_RETURN_ON_ERROR(i2c_master_bus_add_device(s_max30101_bus, &dev_cfg, &s_max30101), TAG, "Device add failed");
309
310 // Configure INIT pin
311 gpio_config_t io_conf = {
312 .pin_bit_mask = (1ULL << MAX30101_INIT_GPIO),
313 .mode = GPIO_MODE_OUTPUT,
314 .pull_up_en = GPIO_PULLUP_DISABLE,
315 .pull_down_en = GPIO_PULLDOWN_DISABLE,
316 .intr_type = GPIO_INTR_DISABLE,
317 };
318 ESP_RETURN_ON_ERROR(gpio_config(&io_conf), TAG, "GPIO config failed");
319 gpio_set_level(MAX30101_INIT_GPIO, 1); // Enable sensor
320 vTaskDelay(pdMS_TO_TICKS(10));
321
322 // Soft reset
323 ESP_RETURN_ON_ERROR(max30101_write(REG_MODE_CONFIG, 0x40), TAG, "Reset failed");
324 vTaskDelay(pdMS_TO_TICKS(10));
325
326 // Configure interrupts
327 max30101_write(REG_INTR_ENABLE_1, 0x40); // PPG_RDY
328 max30101_write(REG_INTR_ENABLE_2, 0x00);
329
330 // FIFO configuration
331 max30101_write(REG_FIFO_CONFIG, 0x00);
332 max30101_write(REG_FIFO_WR_PTR, 0x00);
333 max30101_write(REG_OVF_COUNTER, 0x00);
334 max30101_write(REG_FIFO_RD_PTR, 0x00);
335
336 // SpO2 mode (RED + IR)
337 max30101_write(REG_MODE_CONFIG, 0x03);
338
339 // SpO2 config: ADC=4096nA, 100sps, 411us pulse width
340 max30101_write(REG_SPO2_CONFIG, 0x3B);
341
342 // LED currents
343 max30101_write(REG_LED1_PA, 0x24); // RED
344 max30101_write(REG_LED2_PA, 0x24); // IR
345
346 ESP_LOGI(TAG, "MAX30101 initialized successfully");
347 return ESP_OK;
348}
