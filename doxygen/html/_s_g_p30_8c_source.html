79esp_err_t sgp30_read_values(uint16_t *eco2, uint16_t *tvoc)
80{
81 uint8_t cmd[2] = { 0x20, 0x08 }; // Measure Air Quality command
82 uint8_t data[6];
83
84 // SGP30 needs 12ms to complete measurement
85 // Use longer timeout when BLE is active (up to 2 seconds)
86 esp_err_t err = i2c_master_transmit(s_sgp30, cmd, sizeof(cmd), 2000);
87 if (err != ESP_OK) {
88 s_consecutive_errors++;
89 ESP_LOGW(TAG, "Failed to send measure command: %d (error count: %d)", err, s_consecutive_errors);
90
91 // Attempt recovery after multiple failures
92 if (s_consecutive_errors >= MAX_CONSECUTIVE_ERRORS) {
93 sgp30_reinit_sensor();
94 }
95 return err;
96 }
97
98 // Wait for measurement to complete
99 // Use longer delay when system is busy (BLE active)
100 vTaskDelay(pdMS_TO_TICKS(25)); // Increased from 15ms to 25ms
101
102 // Read the result with retries
103 int retries = 3;
104 while (retries > 0) {
105 err = i2c_master_receive(s_sgp30, data, sizeof(data), 2000);
106 if (err == ESP_OK) {
107 break;
108 }
109 retries--;
110 if (retries > 0) {
111 vTaskDelay(pdMS_TO_TICKS(5)); // Short delay between retries
112 }
113 }
114
115 if (err != ESP_OK) {
116 s_consecutive_errors++;
117 ESP_LOGW(TAG, "Failed to read values after retries: %d (error count: %d)", err, s_consecutive_errors);
118
119 // Attempt recovery after multiple failures
120 if (s_consecutive_errors >= MAX_CONSECUTIVE_ERRORS) {
121 sgp30_reinit_sensor();
122 }
123 return err;
124 }
125
126 // Success - reset error counter
127 s_consecutive_errors = 0;
128
129 // Parse eCO2 and TVOC (each 2 bytes + CRC)
130 if (eco2) {
131 *eco2 = ((uint16_t)data[0] << 8) | data[1];
132 }
133 if (tvoc) {
134 *tvoc = ((uint16_t)data[3] << 8) | data[4];
135 }
136 return ESP_OK;
137}
