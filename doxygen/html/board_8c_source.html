27void board_led_operation(uint8_t pin, uint8_t onoff)
28{
29 for (int i = 0; i < 3; i++) {
30 if (led_state[i].pin != pin) {
31 continue;
32 }
33 if (onoff == led_state[i].previous) {
34 ESP_LOGW(TAG, "led %s is already %s",
35 led_state[i].name, (onoff ? "on" : "off"));
36 return;
37 }
38 gpio_set_level(pin, onoff);
39 led_state[i].previous = onoff;
40 return;
41 }
42
43 ESP_LOGE(TAG, "LED is not found!");
44}
