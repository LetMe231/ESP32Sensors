31esp_err_t aht20_init(void)
32{
33 // Initialize I2C bus
34 i2c_master_bus_config_t bus_cfg = {
35 .clk_source = I2C_CLK_SRC_DEFAULT,
36 .i2c_port = AHT20_I2C_PORT,
37 .scl_io_num = AHT20_SCL_GPIO,
38 .sda_io_num = AHT20_SDA_GPIO,
39 .glitch_ignore_cnt = 7,
40 .flags.enable_internal_pullup = true,
41 };
42
43 ESP_RETURN_ON_ERROR(i2c_new_master_bus(&bus_cfg, &s_aht20_bus), TAG, "I2C bus init failed");
44
45 // Add AHT20 device
46 i2c_device_config_t dev_cfg = {
47 .dev_addr_length = I2C_ADDR_BIT_LEN_7,
48 .device_address = AHT20_ADDR,
49 .scl_speed_hz = AHT20_FREQ_HZ,
50 };
51
52 ESP_RETURN_ON_ERROR(i2c_master_bus_add_device(s_aht20_bus, &dev_cfg, &s_aht20), TAG, "Device add failed");
53
54 // Give sensor time after power-up
55 vTaskDelay(pdMS_TO_TICKS(120));
56
57 // Check calibration status
58 uint8_t cmd = 0x71;
59 uint8_t status = 0;
60 esp_err_t err = i2c_master_transmit_receive(s_aht20, &cmd, 1, &status, 1, 100);
61 if (err != ESP_OK) return err;
62
63 // If not calibrated, send initialization command
64 if ((status & 0x18) != 0x18) {
65 uint8_t init_cmd[3] = { 0xBE, 0x08, 0x00 };
66 err = i2c_master_transmit(s_aht20, init_cmd, sizeof(init_cmd), 100);
67 if (err != ESP_OK) return err;
68 vTaskDelay(pdMS_TO_TICKS(10));
69 }
70
71 ESP_LOGI(TAG, "AHT20 initialized successfully");
72 return ESP_OK;
73}

78esp_err_t aht20_read_x100(int16_t *temp_x100, uint16_t *rh_x100)
79{
80 if (!s_aht20) {
81 return ESP_ERR_INVALID_STATE;
82 }
83
84 // Trigger measurement: 0xAC 0x33 0x00
85 uint8_t trig[3] = { 0xAC, 0x33, 0x00 };
86 ESP_RETURN_ON_ERROR(i2c_master_transmit(s_aht20, trig, sizeof(trig), 100), TAG, "Trigger failed");
87
88 // Wait for conversion (~80ms)
89 vTaskDelay(pdMS_TO_TICKS(85));
90
91 // Read 6 bytes (status + 5 data bytes)
92 uint8_t d[6] = {0};
93 ESP_RETURN_ON_ERROR(i2c_master_receive(s_aht20, d, sizeof(d), 100), TAG, "Read failed");
94
95 // If busy bit still set, retry
96 for (int i = 0; (d[0] & 0x80) && i < 3; i++) {
97 vTaskDelay(pdMS_TO_TICKS(20));
98 ESP_RETURN_ON_ERROR(i2c_master_receive(s_aht20, d, sizeof(d), 100), TAG, "Read retry failed");
99 }
100 if (d[0] & 0x80) return ESP_ERR_TIMEOUT;
101
102 // Parse 20-bit humidity and temperature values
103 uint32_t raw_rh = ((uint32_t)d[1] << 12) | ((uint32_t)d[2] << 4) | ((uint32_t)d[3] >> 4);
104 uint32_t raw_t = (((uint32_t)d[3] & 0x0F) << 16) | ((uint32_t)d[4] << 8) | (uint32_t)d[5];
105
106 // Convert to physical values using datasheet formulas
107 // RH = (SRH / 2^20) * 100%
108 // T = (ST / 2^20) * 200 - 50°C
109 float rh = (raw_rh * 100.0f) / 1048576.0f;
110 float t = (raw_t * 200.0f) / 1048576.0f - 50.0f;
111
112 // Convert to fixed-point (x100)
113 int32_t tx = (int32_t)lroundf(t * 100.0f);
114 int32_t hx = (int32_t)lroundf(rh * 100.0f);
115
116 // Clamp values
117 if (tx < -4000) tx = -4000;
118 if (tx > 8500) tx = 8500;
119 if (hx < 0) hx = 0;
120 if (hx > 10000) hx = 10000;
121
122 *temp_x100 = (int16_t)tx;
123 *rh_x100 = (uint16_t)hx;
124
125 return ESP_OK;
126}
